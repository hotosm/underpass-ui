## Install

```bash
npm install --save underpass-ui-dev
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'underpass-ui-dev'
import 'underpass-ui-dev/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
